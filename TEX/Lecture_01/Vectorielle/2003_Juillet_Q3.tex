  \begin{tcolorbox}[basic] \smallskip
  Thèse : $|\vect{AC}|^2 + |\vect{BD}|^2 + 4|\vect{EF}|^2=|\vect{AB}|^2 + |\vect{BC}|^2 + |\vect{CD}|^2 + |\vect{DA}|^2 $. 
  \end{tcolorbox} \smallskip

  	\heart Le produit scalaire est commutatif et distributif. 
	 \medskip
  \begin{align*}
    &|\vect{AC}|^2 + |\vect{BD}|^2 + 4|\vect{EF}|^2 \\[0.5em]
    &= \vect{AC}\cdot\vect{AC} + \vect{BD}\cdot\vect{BD} + 4\vect{EF}\cdot\vect{EF}, \\[0.5em]
       &= (\vect{AB} + \vect{BC})\cdot(\vect{AB} + \vect{BC}) + (\vect{BC} + \vect{CD})\cdot(\vect{BC} + \vect{CD})  \\ &\hspace{60mm} + 4(-\frac{\vect{BC}}{2} + \frac{\vect{AD}}{2})\cdot(-\frac{\vect{BC}}{2}+ \frac{\vect{AD}}{2}), \\[0.5em]
    &=\vect{AB}\cdot\vect{AB}+3\vect{BC}\cdot\vect{BC}+\vect{CD}\cdot\vect{CD}+\vect{AD}\cdot\vect{AD}  \\ &\hspace{60mm} + 2(\vect{BC}\cdot\vect{CD} -\vect{BC}\cdot\vect{AD} + \vect{BC}\cdot\vect{AB}), \\[0.5em]
      &=|\vect{AB}|^2+3|\vect{BC}|^2+|\vect{CD}|^2+|\vect{AD}|^2 + 2(\vect{BC}\cdot\vect{CD} -\vect{BC}\cdot\vect{AD} + \vect{BC}\cdot\vect{AB}), \\[0.5em]
  &=|\vect{AB}|^2+3|\vect{BC}|^2+|\vect{CD}|^2+|\vect{AD}|^2 + 2\vect{BC}\cdot( \vect{CD} + \vect{DA} + \vect{AB}), \\[0.5em]
    &=|\vect{AB}|^2+3|\vect{BC}|^2+|\vect{CD}|^2+|\vect{AD}|^2 + 2\vect{BC}\cdot\vect{CB}, \\[0.5em]   
  &=|\vect{AB}|^2+|\vect{BC}|^2+|\vect{CD}|^2+|\vect{AD}|^2.  \hspace{50mm} \qed
  \end{align*}

	  \underline{Résumé}\\ \flushleft
          \begin{itemize}
	  \item Montrer l'égalité d'une expression vectorielle = décomposer vecteurs d'un membre avec les vecteurs du second membre.
	  \item $|\vect{AB}|^2=\vect{AB}\cdot\vect{AB}$.
            \item  Le produit scalaire est commutatif et distributif.
