 \def\enonce{  \frametitle{Q4 Septembre 2002.}
	 
	  Soit $ABCD$ un tétraèdre. On note respectivement $h_A$ et $h_B$ les hauteurs du
	  tétraèdre issues de $A$ et $B$.
	  \begin{enumerate}
             \renewcommand{\theenumi}{\alph{enumi})}
	   \item Démontrer que $h_A$ et $h_B$ sont concourantes si et seulement si la droite $AB$ est
		orthogonale à la droite $CD$.
	   \item Démontrer que les hauteurs d’un tétraèdre sont concourantes si les arêtes
		 opposées de ce tétraèdre sont orthogonales deux à deux. 
	  \end{enumerate}
  }
  \def\hypotheses{  \underline{Hypothèses} 
		      \begin{enumerate}
		      \item $AA'\bot BCD$,
		      \item $BB'\bot ACD$.
		      \end{enumerate} 
  }
  \def\these{  \underline{Thèse} 
    \begin{enumerate}
       \renewcommand{\theenumi}{\alph{enumi})}
		       \item $AA'\nmid BB' \leftrightarrow CD \orthog AB$,
		       \item côtés opposés $\orthog \rightarrow$ hauteurs concourantes.
		      \end{enumerate}
}

	  \underline{Résumé}\\ \flushleft
          \begin{itemize}
          \item Si \textit{ssi}, montrer les 2 sens de l'énoncé.
          \item Une droite $\bot$ à un plan est $\orthog$ à toutes les droites de ce plan.
          \item Il n'existe qu'un plan $\bot$ à une droite donnée passant par un point donné.
          \item  2 droites sont $\nmid$ si elles $\in$ au même plan et ne sont pas $\parallel$.
            \item  Trois droites de directions différentes et sécantes 2 à 2 sont concourantes (à prouver).
          \end{itemize}
